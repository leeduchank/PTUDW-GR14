-- Create the "spaces" table
CREATE TABLE spaces (
    id_spaces INT PRIMARY KEY,
    address VARCHAR(255),
    long_s FLOAT,
    lat_s FLOAT,
    type VARCHAR(50),
    format VARCHAR(50),
    ward VARCHAR(50)
);

-- Create the "surfaces" table
CREATE TABLE surfaces (
    id_surfaces INT PRIMARY KEY,
    format VARCHAR(50),
    width INT,
    height INT,
    imgURL VARCHAR(255),
    space INT,
    FOREIGN KEY (space) REFERENCES spaces(id_spaces)
);

-- Create the "report" table
CREATE TABLE report (
    id_reports INT PRIMARY KEY,
    address VARCHAR(255),
    long_r FLOAT,
    lat_r FLOAT,
    report_date DATETIME,
    content TEXT,
    email VARCHAR(255),
    phone VARCHAR(20),
    state INT,
    surface INT,
    FOREIGN KEY (surface) REFERENCES surfaces(id_surfaces)
);
